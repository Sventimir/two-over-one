\label{subsec:1nt-op}

\ctr{1}{NT} --- ?
\begin{itemize}
  \item \ctr{2}{\c} -- \conv{Stayman} \ask{\suit{4+}{\major}} or any \inv\ hand
  \item \ctr{2}{\d} -- \conv{Texas}: \suit{5+}{\h}, any strength
  \item \ctr{2}{\h} -- \conv{Texas}: \suit{5+}{\s}, any strength
  \item \ctr{2}{\s} -- \conv{Minor Stayman} \ask{\suit{4+}{\minor}}, slam aspirations
  \item \ctr{2}{NT} -- \conv{Texas}: \suit{5+}{\c}, any strength
  \item \ctr{3}{\c} -- \conv{Texas}: \suit{5+}{\d}, any strength
  \item \ctr{3}{\d/\major} -- ?
  \item \ctr{3}{NT} -- \nat\/ to play
  \item \ctr{4}{\c} -- \conv{Gerber}: \ask{aces} (\ctr{4}{\d} = 0/3; \ctr{4}{\h} = 1/4 etc.)
  \item \ctr{4}{\d} -- \conv{Texas}: \suit{6+}{\h} moderate strength, no slam aspirations
  \item \ctr{4}{\h} -- \conv{Texas}: \suit{6+}{\s} moderate strength, no slam aspirations
\end{itemize}
