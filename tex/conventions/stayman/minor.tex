\subsection{Minor Stayman}
\label{subsec:minor-stayman}

\begin{flushleft}
  \ctr{1}{NT} --- \ctr{2}{\s} --- ? \linebreak
  This bidding promises strong, minor-oriented hand with slam aspirations.
\end{flushleft}
\begin{itemize}
  \item \ctr{2}{NT} -- \suit{4\hyp 5}{\c} \& \suit{4\hyp 5}{\d}
  \item \ctr{3}{\c} -- \suit{4}{\c}, minor slam interst
  \item \ctr{3}{\d} -- \suit{4}{\d}, minor slam interst
  \item \ctr{3}{\h} -- \suit{5}{\c}, minor slam interst
  \item \ctr{3}{\s} -- \suit{5}{\d}, minor slam interst
  \item \ctr{3}{NT} -- $2\hyp 3$ cards in both minors, \no\ slam interest.
\end{itemize}
